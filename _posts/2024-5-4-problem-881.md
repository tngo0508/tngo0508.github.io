## Stack Approach - TLE

```python
class Solution:
    def numRescueBoats(self, people: List[int], limit: int) -> int:
        people.sort()
        res = 0
        while people:
            curr_weight = people.pop()
            num_of_people = 1
            stack = []
            while people and curr_weight < limit and num_of_people < 3:
                weight = people.pop()
                if curr_weight + weight <= limit:
                    break
                stack.append(weight)

            while stack:
                people.append(stack.pop())

            res += 1

        return res
```
